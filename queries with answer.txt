

-- 1. Yearly Avg Price
SELECT EXTRACT(YEAR FROM Date) AS Year,
ROUND(AVG(Close_price), 2) AS Avg_Price
FROM bitcoin
GROUP BY Year
ORDER BY Year;

-- 2. All Time High
SELECT Date, High_price
FROM bitcoin
ORDER BY High_price DESC
LIMIT 1;

-- 3. All Time Low
SELECT Date, Low_price
FROM bitcoin
ORDER BY Low_price ASC
LIMIT 1;

-- 4. Bull vs Bear Days
SELECT
CASE
    WHEN Close_price > Open_price THEN 'Bull'
    ELSE 'Bear'
END AS Market_Type,
COUNT(*) AS Total_Days
FROM bitcoin
GROUP BY Market_Type;

-- 5. Highest Volume Days
SELECT Date, Close_price, Volume
FROM bitcoin
ORDER BY Volume DESC
LIMIT 10;

-- 6. Window Function — Running Avg Price
SELECT Date, Close_price,
ROUND(AVG(Close_price) OVER (
    ORDER BY Date
    ROWS BETWEEN 29 PRECEDING AND CURRENT ROW
), 2) AS Moving_Avg_30Day
FROM bitcoin
ORDER BY Date;


-- 7. CTE — Yearly Stats
WITH YearlyStats AS (
    SELECT
    EXTRACT(YEAR FROM Date) AS Year,
    ROUND(AVG(Close_price), 2) AS Avg_Price,
    ROUND(MAX(High_price), 2) AS Year_High,
    ROUND(MIN(Low_price), 2) AS Year_Low,
    ROUND(AVG(High_price - Low_price), 2) AS Avg_Volatility
    FROM bitcoin
    GROUP BY Year
)
SELECT *,
ROUND(Avg_Price - LAG(Avg_Price) 
OVER (ORDER BY Year), 2) AS YoY_Change,
ROUND((Avg_Price - LAG(Avg_Price) 
OVER (ORDER BY Year)) / 
LAG(Avg_Price) 
OVER (ORDER BY Year) * 100, 2) AS Growth_Pct
FROM YearlyStats
ORDER BY Year;


-- 8. RANK — Best Performing Years
SELECT
EXTRACT(YEAR FROM Date) AS Year,
ROUND(AVG(Close_price), 2) AS Avg_Price,
RANK() OVER (
    ORDER BY AVG(Close_price) DESC
) AS Price_Rank
FROM bitcoin
GROUP BY Year
ORDER BY Price_Rank;


-- 9. Most Volatile Days — Percentile
SELECT Date, Close_price,
ROUND((High_price - Low_price)::numeric, 2) AS Price_Range,
ROUND((PERCENT_RANK() OVER (
    ORDER BY High_price - Low_price
) * 100)::numeric, 2) AS Volatility_Percentile
FROM bitcoin
ORDER BY Price_Range DESC
LIMIT 10;

-- 10. Monthly Seasonality Analysis
SELECT
TO_CHAR(Date, 'Month') AS Month_Name,
EXTRACT(MONTH FROM Date) AS Month_Num,
ROUND(AVG(Close_price), 2) AS Avg_Price,
ROUND(AVG(Volume), 0) AS Avg_Volume,
ROUND(AVG(High_price - Low_price), 2) AS Avg_Volatility,
COUNT(CASE WHEN Close_price > Open_price 
    THEN 1 END) AS Bull_Days,
COUNT(CASE WHEN Close_price <= Open_price
    THEN 1 END) AS Bear_Days
FROM bitcoin
GROUP BY Month_Name, Month_Num
ORDER BY Month_Num;
